## 3. Configuración de Zsh Modular (.config/zsh/plugins.zsh)

Para evitar frameworks pesados, utiliza una función de carga manual para los 4 plugins recomendados:

Para configurar todo automáticamente, abre tu terminal Ghostty y pega el siguiente comando:
curl -sSL https://raw.githubusercontent.com/tu-usuario/repo/main/setup.sh | bash

```
#!/bin/bash

# 1. Crear estructura de directorios XDG y carpetas de estado
mkdir -p ~/.config/zsh/plugins
mkdir -p ~/.cache/zsh
mkdir -p ~/.local/state/zsh
mkdir -p ~/.local/bin

# 2. Configurar el entorno inicial en /etc/zsh/zshenv. Establecer ZDOTDIR de forma global (requiere sudo)
echo "Configurando ZDOTDIR en /etc/zsh/zshenv..."
sudo bash -c 'cat <<EOF > /etc/zsh/zshenv
export XDG_CONFIG_HOME="\$HOME/.config"
export ZDOTDIR="\$XDG_CONFIG_HOME/zsh"
EOF'

# 3. Crear .zshenv (Variables de entorno y Path)
cat <<EOF > ~/.config/zsh/.zshenv
export XDG_CACHE_HOME="\$HOME/.cache"
export XDG_DATA_HOME="\$HOME/.local/share"
export XDG_STATE_HOME="\$HOME/.local/state"
export EDITOR="nvim"
export PAGER="bat"
export PATH="\$HOME/.local/bin:\$PATH"
export STARSHIP_CONFIG="\$ZDOTDIR/starship.toml"
EOF

# 4. Crear .zshrc principal con carga modular
cat <<EOF > ~/.config/zsh/.zshrc
# Configuración de Historial
HISTFILE="\$XDG_STATE_HOME/zsh/history"
HISTSIZE=100000
SAVEHIST=100000
setopt SHARE_HISTORY
setopt HIST_IGNORE_ALL_DUPS
setopt HIST_IGNORE_SPACE
setopt AUTO_CD

# Inicialización de herramientas
eval "\$(zoxide init zsh)"
eval "\$(starship init zsh)"

# Autocompletado profesional
autoload -Uz compinit && compinit -d "\$XDG_CACHE_HOME/zsh/zcompdump"
zstyle ':completion:*' menu select
zstyle ':completion:*' matcher-list 'm:{a-zA-Z}={A-Za-z}'

# Carga de archivos modulares
source "\$ZDOTDIR/aliases.zsh"
source "\$ZDOTDIR/bindings.zsh"
source "\$ZDOTDIR/fzf.zsh"
source "\$ZDOTDIR/plugins.zsh"
EOF

# 5. Configurar FZF con vistas previas
cat <<EOF > ~/.config/zsh/fzf.zsh
# Configuración para Mac con Homebrew [12]
source <(fzf --zsh)
export FZF_DEFAULT_OPTS="--height 40% --layout=reverse --border --preview 'bat --style=numbers --color=always {} | head -500'"
EOF

# 6. Configurar los 4 Plugins (Sistema de autogestión sin frameworks)
cat <<EOF > ~/.config/zsh/plugins.zsh
function zplug() {
  PLUGIN_NAME=\$(echo \$1 | cut -d/ -f2)
  PLUGIN_DIR="\$ZDOTDIR/plugins/\$PLUGIN_NAME"
  if [ ! -d "\$PLUGIN_DIR" ]; then
    echo "Instalando plugin: \$PLUGIN_NAME..."
    git clone --depth 1 "https://github.com/\$1.git" "\$PLUGIN_DIR"
  fi
  source "\$PLUGIN_DIR/\$PLUGIN_NAME.plugin.zsh" 2>/dev/null || source "\$PLUGIN_DIR/\$PLUGIN_NAME.zsh"
}

# Plugins recomendados
zplug zsh-users/zsh-syntax-highlighting
zplug zsh-users/zsh-autosuggestions
zplug zsh-users/zsh-history-substring-search
zplug jeffreytse/zsh-vi-mode
EOF [1]

# 7. Atajos de teclado (Bindings) para Substring Search y Vi-Mode
cat <<EOF > ~/.config/zsh/bindings.zsh
# Búsqueda con flechas arriba/abajo usando substring-search
bindkey '^[[A' history-substring-search-up
bindkey '^[[B' history-substring-search-down

# Configuración de cursor para Vi-Mode
ZSH_VI_MODE_CURSOR_SECTION_NAME="cursor"
ZSH_VI_MODE_INSERT_CURSOR=beam
ZSH_VI_MODE_NORMAL_CURSOR=block
EOF [14-16]

# 8. Alias mejorados con herramientas modernas
cat <<EOF > ~/.config/zsh/aliases.zsh
alias ls="eza --icons"
alias ll="eza -l --icons --git"
alias la="eza -ah --icons --git"
alias cat="bat"
alias grep="rg"
alias vim="nvim"
alias g="git"
EOF

# 9. Instalar herramientas base con Homebrew (macOS)
if command -v brew >/dev/null 2>&1; then
    brew install zoxide fzf starship eza bat fd ripgrep nvim
fi

echo "Configuración completada. Reinicia Ghostty para ver los cambios."
```
